## 226 翻转二叉树

```
public TreeNode invertTree(TreeNode root) {
​    if(root == null)
​      return root;

​    TreeNode node = root.left;
​    root.left = root.right;
​    root.right = node;

​    invertTree(root.left);
​    invertTree(root.right);
​    return root;
 }
```

## 394 字符串解码

```
public String decodeString(String s) {
​    LinkedList<Integer> stack_num = new LinkedList<>();
​    LinkedList<String> stack_word = new LinkedList<>();

​    StringBuilder sb = new StringBuilder();
​    int num=0;

​    for(char ch : s.toCharArray()){
​      if(ch >='0' && ch <='9') 
​        num = num*10 + (ch - '0');
​      else if(ch == '['){
​        stack_num.add(num);
​        stack_word.add(sb.toString());
​        sb.setLength(0);
​        num = 0;
​      }
​      else if(ch == ']'){
​        StringBuilder tmp = new StringBuilder();
​        int cur_multi = stack_num.removeLast();
​        for(int i = 0; i < cur_multi; i++) 
​          tmp.append(sb);
​        sb = new StringBuilder(stack_word.removeLast() + tmp);
​      }
​      else{
​        sb.append(ch);
​      }
​    }
​    return sb.toString();
  }
```

## 316 去除重复字母

```
public String removeDuplicateLetters(String s) {
​    LinkedList<Character> list = new LinkedList<>();

​    for(int i=0;i<s.length();i++){
​      char c = s.charAt(i);
​      if(!list.contains(c)){
​        while(!list.isEmpty() && c < list.getLast() && s.indexOf(list.getLast(),i)!=-1){
​          list.removeLast();
​        }
​        list.add(c);
​      }
​    }
​    StringBuilder sb = new StringBuilder();
​    for(Character ch : list)
​      sb.append(ch);
​    
​    return sb.toString();
  }
```

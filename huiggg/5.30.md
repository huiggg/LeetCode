## 764 最大加号标志

```
/*
通过两次遍历，来构造`dp`数组。
1. 从`左上角`开始遍历，计算每个点`向左和向上`的臂长
2. 从`右下角`开始遍历，计算每个点`向右和向下`的臂长

*/
public int orderOfLargestPlusSign(int N, int[][] mines) {
​    int[][] matrix = new int[N][N];
​    for(int i =0;i<N;i++){
​      for(int j=0;j<N;j++){
​        matrix[i][j] =1;
​      }
​    }
​    for(int[] z:mines){
​      matrix[z[0]][z[1]] = 0;
​    }

​    int[][] up = new int[N][N];
​    int[][] down = new int[N][N];
​    int[][] left = new int[N][N];
​    int[][] right = new int[N][N];

​    //左上
​    for(int i=0;i<N;i++){
​      for(int j=0;j<N;j++){
​        if(matrix[i][j] == 0) continue;
​        up[i][j] = i > 0 ? up[i-1][j] + 1 : 1;
​        left[i][j] = j > 0 ? left[i][j-1] +1 : 1;
​      }
​    }

​    //右下
​    for(int i=N-1;i>=0;i--){
​      for(int j = N-1;j>=0;j--){
​        if(matrix[i][j] == 0) continue;
​        down[i][j] = i < N-1 ? down[i+1][j] +1 : 1;
​        right[i][j] = j < N-1 ? right[i][j+1]+1:1;
​      }
​    }
​    int ans = 0;
​    for(int i=0;i<N;i++){
​      for(int j=0;j<N;j++){
​        int k = Math.min(up[i][j],down[i][j]);
​        int m = Math.min(left[i][j],right[i][j]);
​        k = Math.min(k,m);
​        ans = Math.max(ans,k);
​      }
​    }
 	 return ans;
}
```

public int[] getMaxMatrix(int[][] matrix) {
​    int[] ans = new int[4];
​    int m = matrix.length;
​    int n = matrix[0].length;
​    int max = Integer.MIN_VALUE;
​    int r1 = 0, c1 = 0;
​    int[] arr = new int[n];
​    for (int i = 0; i < m; i ++) {
​      Arrays.fill(arr, 0);
​      for (int j = i; j < m; j ++) {
​        int sum = 0;
​        for (int k = 0; k < n; k ++) {
​          arr[k] += matrix[j][k];
​          if (sum > 0) {
​            sum += arr[k];
​          } else {
​            sum = arr[k];
​            r1 = i;
​            c1 = k;
​          }
​          if (sum > max) {
​            max = Math.max(max, sum);
​            ans[0] = r1;
​            ans[1] = c1;
​            ans[2] = j;
​            ans[3] = k;
​          }
​        }
​      }
​    }
​    return ans;
  }
